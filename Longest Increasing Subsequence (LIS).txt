ll longestIncreasingSubsequence(ll n, ll a[])
{
    if (dp[n] != -1)
        return dp[n];
    ll ans = 1;
    for (int i = 0; i < n; i++)
    {
        if (a[i] > a[n])
        {
            ans = max(ans, longestIncreasingSubsequence(i, a) + 1);
        }
    }
    return dp[n] = ans;
}
