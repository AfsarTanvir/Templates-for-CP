vector<vector<int>> findDivisorsInRange(int start, int end)
{
    vector<vector<int>> allDivisors;
    for (int n = start; n <= end; ++n)
    {
        vector<int> divisors;
        for (int i = 1; i <= n; ++i)
        {
            if (n % i == 0)
            {
                divisors.push_back(i);
            }
        }
        allDivisors.push_back(divisors);
    }
    return allDivisors;
}
