long long binarySearch(vector<int> &nums, int target)
{
    long long low = 0, high = nums.size() - 1, mid;
    while (low <= high)
    {
        mid = (high + low) >> 1;
        if (nums[mid] == target)
        {
            return mid;
        }
        else if (nums[mid] < target)
        {
            low = mid + 1;
        }
        else
        {
            high = mid - 1;
        }
    }
    return -1;
    // return high;
}
